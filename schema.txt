BEGIN;
CREATE TABLE "homes_owner" (                                                                  
  "id" serial NOT NULL PRIMARY KEY,                                                         
  "name" text NOT NULL                                                                      
);                                                                            
CREATE TABLE "homes_house" (                                                                  
  "id" serial NOT NULL PRIMARY KEY,                                                         
  "address" text NOT NULL,                                                                  
  "owner_id" integer NOT NULL REFERENCES "homes_owner" ("id") DEFERRABLE INITIALLY DEFERRED 
);                                                                                           
COMMIT;
